#include <iostream>
#include <vector>
#include <iomanip>

using namespace std;

struct Card{
    int value;
    char cardLetter;
};

class Deck{
    
    vector<Card> cards;
    
    public:
    
        Deck(bool isDiscardDeck){
            if (!isDiscardDeck){
                // constructor to instantly make the cards
                CreateDeck();
                Shuffle();
            }
        }
        
        void CreateDeck(){
            // loop for the suits
            for (int i = 0; i < 4; i++){
                // loop for each card type
                for (int j = 1; j <= 13; j++){
                    // create the card
                    Card card;
                    // give the card values based on point in loop
                    switch (j){
                        case 1:
                            card.cardLetter = 'A';
                            card.value = 11;
                            cards.push_back(card);
                            break;
                        case 2:
                            card.cardLetter = '2';
                            card.value = 2;
                            cards.push_back(card);
                            break;
                        case 3:
                            card.cardLetter = '3';
                            card.value = 3;
                            cards.push_back(card);
                            break;
                        case 4:
                            card.cardLetter = '4';
                            card.value = 4;
                            cards.push_back(card);
                            break;
                        case 5:
                            card.cardLetter = '5';
                            card.value = 5;
                            cards.push_back(card);
                            break;
                        case 6:
                            card.cardLetter = '6';
                            card.value = 6;
                            cards.push_back(card);
                            break;
                        case 7:
                            card.cardLetter = '7';
                            card.value = 7;
                            cards.push_back(card);
                            break;
                        case 8:
                            card.cardLetter = '8';
                            card.value = 8;
                            cards.push_back(card);
                            break;
                        case 9:
                            card.cardLetter = '9';
                            card.value = 9;
                            cards.push_back(card);
                            break;
                        case 10:
                            card.cardLetter = 'X';
                            card.value = 10;
                            cards.push_back(card);
                            break;
                        case 11:
                            card.cardLetter = 'J';
                            card.value = 10;
                            cards.push_back(card);
                            break;
                        case 12:
                            card.cardLetter = 'Q';
                            card.value = 10;
                            cards.push_back(card);
                            break;
                        case 13:
                            card.cardLetter = 'K';
                            card.value = 10;
                            cards.push_back(card);
                            break;
                    }
                }
            }
        }
        
        void Shuffle(){
            srand(time(NULL));
            // loop to shuffle cards
            for (int i = cards.size() - 1; i >= 0; i--){
                int j = rand() % (i + 1);
                swap(cards[i], cards[j]);
            }
            // second time for fun
            for (int i = cards.size() - 1; i >= 0; i--){
                int j = rand() % (i + 1);
                swap(cards[i], cards[j]);
            }
        }
        
        // print the letter of cards in deck
        void PrintCards(){
            if (cards.empty()) return;
            for (int i = 0; i < cards.size(); i++){
                cout << cards[i].cardLetter << endl;
            }
        }
        
        int getSizeOfDeck(){
            return cards.size();
        }
        
        Card Deal(){
            // give out cards
            Card toDeal;
            toDeal = cards.back();
            cards.pop_back();
            return toDeal;
        }
        
        bool CheckIfEmpty(){
            return cards.empty();
        }
        
        void AddToDiscardDeck(Card card){
            cards.push_back(card);
        }
};

class Hand{
    
    // int for total value of hand
    int handSum;
    bool isHouse = false;
    bool isHouseStartingHand = false;
    // dynamic arrays for splits
    vector<Card> cardsInHand;
    vector<Card> cardsInHand2;
    
    public:
        Hand(bool isTheHouse){
            // initialize hand to 0
            handSum = 0;
            isHouse = isTheHouse;
            isHouseStartingHand = isTheHouse;
        }
        
        void DisplayHand(){
            if (cardsInHand.empty()) return;
            if (isHouseStartingHand) {
                cout << cardsInHand[0].cardLetter << " ?";
                return;
            }
            for (int i = 0; i < cardsInHand.size(); i++){
                cout << cardsInHand[i].cardLetter << " ";
            }
            if (!cardsInHand2.empty()){
                cout << "  ";
                for (int i = 0; i < cardsInHand.size(); i++){
                    cout << cardsInHand2[i].cardLetter << " ";
                }
            }
        }
        
        void HitMe(Deck &deck){
            // need to add later to grab from other deck
            if (deck.getSizeOfDeck() == 0) return;
            // add card to cards in hand
            cardsInHand.push_back(deck.Deal());
            if (isHouseStartingHand){
                handSum = cardsInHand[0].value;
                return;
            }
            // reset value of hand
            handSum = 0;
            // calculate value in hand
            for (int i = 0; i < cardsInHand.size(); i++){
                handSum += cardsInHand[i].value;
            }
        }
        
        void Split(){
            if (cardsInHand[0].cardLetter == cardsInHand[1].cardLetter){
                cardsInHand2.push_back(cardsInHand[1]);
                cardsInHand.pop_back();
            }
        }
        
        int GetHandSum(){
            if (!isHouseStartingHand){
                handSum = 0;
                for (int i = 0; i < cardsInHand.size(); i++){
                    handSum += cardsInHand[i].value;
                }
            }
            return handSum;
        }
        
        // for printing whole house hand
        void NotStartingHand(){
            isHouseStartingHand = false;
        }
        
        // alters variable to hide second card of house
        void IsStartingHand(){
            isHouseStartingHand = true;
        }
        
        vector<Card> GetCardsInHand(int hand){
            if (hand == 1) return cardsInHand;
            else return cardsInHand2;
        }
        
        // empties hand
        void ClearHand(){
            cardsInHand.clear();
            if (!cardsInHand2.empty()){
                cardsInHand2.clear();
            }
        }
};

void ShowTable(Hand &playerHand, Hand &houseHand, int money){
    cout << setw(17) << "House: " << houseHand.GetHandSum() << endl;
    cout << setw(30) << setfill('#');
    cout << "#";
    cout << "\n";
    cout << setw(13) << setfill(' ');
    houseHand.DisplayHand();
    cout << "\n\n\n";
    cout << setw(13) << setfill(' ');
    playerHand.DisplayHand();
    cout << "\n";
    cout << setw(30) << setfill('#');
    cout << "#" << endl;
    cout << setw(18) << setfill(' ');
    cout << "Player: " << playerHand.GetHandSum() << endl;
    cout << "Money: " << money;
}

bool CheckForWin(int playerSum, int houseSum){
    if (playerSum > 21){
        cout << "You lose.\n\n";
        return false;
    }
    else if (houseSum > 21){
        cout << "YOU WIN!!!\n\n";
        return true;
    }
    else if (playerSum > houseSum){
        cout << "YOU WIN!!!\n\n";
        return true;
    }
    else if (houseSum > playerSum){
        cout << "You lose.\n\n";
        return false;
    }
    return false;
}

int main() {
    int money = 1000;
    int bet;
    // loop flag
    bool running = true;
    bool playerStay = false;
    bool playerContinue = true;
    // false = deck1, true = deck2
    bool deckToGrabFrom = false;
    // input command
    char com;
    char playerContinueAns;
    // not the discard deck
    Deck deck1(false);
    // discard deck
    Deck deck2(true);
    // not the house hand
    Hand playerHand(false);
    int playerSum;
    // is the house hand
    Hand houseHand(true);
    int houseSum;
    // starting hands
    playerHand.HitMe(deck1);
    houseHand.HitMe(deck1);
    playerHand.HitMe(deck1);
    houseHand.HitMe(deck1);
    
    ShowTable(playerHand, houseHand, money);
    
    while (playerContinue){
        
        cout << endl;
        cout << "Enter a bet amount: ";
        cin >> bet;
        
        money -= bet;
        while (true){
            if (deck1.CheckIfEmpty() == true && deckToGrabFrom == false){
                // swap to the second deck for grabbing
                deckToGrabFrom = true;
                deck2.Shuffle();
            } 
            else if (deck2.CheckIfEmpty() == true && deckToGrabFrom == true){
                // swap back to the first deck for grabbing
                deckToGrabFrom = false;
                deck1.Shuffle();
            }
            
            cout << "\n";
            
            cout << "Please enter a move (h - hit, s - stay, p - split): ";
            cin >> com;
            
            switch (com){
                case 'h':
                    if (deckToGrabFrom == false){
                        playerHand.HitMe(deck1);
                    } 
                    else {
                        playerHand.HitMe(deck2);
                    }
                    break;
                case 's':
                    playerStay = true;
                    running = false;
                    houseHand.NotStartingHand();
                    break;
                case 'p':
                    playerHand.Split();
                    break;
                default:
                    cout << "\nProper input not detected.\n\n";
            }
            
            playerSum = playerHand.GetHandSum();
            houseSum = houseHand.GetHandSum();
            
            if (playerSum > 21) break;
            
            if (playerStay == true) break;
            else if (running == false) break;
            
            ShowTable(playerHand, houseHand, money);
        }
        
        if (playerStay == true){
            // loop for house to try to beat player
            while(houseSum <= playerSum && houseSum != 21){
                if (deck1.CheckIfEmpty() == false){
                    houseHand.HitMe(deck1);
                } 
                else {
                    houseHand.HitMe(deck2);
                }
                houseSum = houseHand.GetHandSum();
            }
        }
        
        cout << endl;
        if (CheckForWin(playerSum, houseSum) == true) {
            money += (bet * 2);
        }
        ShowTable(playerHand, houseHand, money);
        
        if (money <= 0) break;
        
        cout << endl;
        cout << "Would you like to play again? (y/n): ";
        cin >> playerContinueAns;
        
        if (playerContinueAns == 'n'){
            playerContinue = false;
        }
        else {
            running = true;
            playerStay = false;
            houseHand.IsStartingHand();
            vector<Card> temp = playerHand.GetCardsInHand(1);
            if (!temp.empty()){
                // need to reset hands and add to the discard deck
                for (int i = temp.size() - 1; i >= 0; i--){
                    deck2.AddToDiscardDeck(temp[i]);
                }
            }
            temp = playerHand.GetCardsInHand(2);
            if (!temp.empty()){
                // need to reset hands and add to the discard deck
                for (int i = temp.size() - 1; i >= 0; i--){
                    deck2.AddToDiscardDeck(temp[i]);
                }
            }
            playerHand.ClearHand();
            houseHand.ClearHand();
            if (deck1.CheckIfEmpty() == false){
                playerHand.HitMe(deck1);
                playerHand.HitMe(deck1);
            } 
            else {
                playerHand.HitMe(deck2);
                playerHand.HitMe(deck2);
            }
            
            if (deck1.CheckIfEmpty() == false){
                houseHand.HitMe(deck1);
                houseHand.HitMe(deck1);
            } 
            else {
                houseHand.HitMe(deck2);
                houseHand.HitMe(deck2);
            }
            ShowTable(playerHand, houseHand, money);
        }
    }
    

    return 0;
}
