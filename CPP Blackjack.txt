using namespace std;

struct Card{
    int value;
    char cardLetter;
};

class Deck{
    
    vector<Card> cards;
    
    public:
    
        Deck(bool isDiscardDeck){
            if (!isDiscardDeck){
                // constructor to instantly make the cards
                CreateDeck();
                Shuffle();
            }
        }
        
        void CreateDeck(){
            // loop for the suits
            for (int i = 0; i < 4; i++){
                // loop for each card type
                for (int j = 1; j <= 13; j++){
                    // create the card
                    Card card;
                    // give the card values based on point in loop
                    switch (j){
                        case 1:
                            card.cardLetter = 'A';
                            card.value = 11;
                            cards.push_back(card);
                            break;
                        case 2:
                            card.cardLetter = '2';
                            card.value = 2;
                            cards.push_back(card);
                            break;
                        case 3:
                            card.cardLetter = '3';
                            card.value = 3;
                            cards.push_back(card);
                            break;
                        case 4:
                            card.cardLetter = '4';
                            card.value = 4;
                            cards.push_back(card);
                            break;
                        case 5:
                            card.cardLetter = '5';
                            card.value = 5;
                            cards.push_back(card);
                            break;
                        case 6:
                            card.cardLetter = '6';
                            card.value = 6;
                            cards.push_back(card);
                            break;
                        case 7:
                            card.cardLetter = '7';
                            card.value = 7;
                            cards.push_back(card);
                            break;
                        case 8:
                            card.cardLetter = '8';
                            card.value = 8;
                            cards.push_back(card);
                            break;
                        case 9:
                            card.cardLetter = '9';
                            card.value = 9;
                            cards.push_back(card);
                            break;
                        case 10:
                            card.cardLetter = 'X';
                            card.value = 10;
                            cards.push_back(card);
                            break;
                        case 11:
                            card.cardLetter = 'J';
                            card.value = 10;
                            cards.push_back(card);
                            break;
                        case 12:
                            card.cardLetter = 'Q';
                            card.value = 10;
                            cards.push_back(card);
                            break;
                        case 13:
                            card.cardLetter = 'K';
                            card.value = 10;
                            cards.push_back(card);
                            break;
                    }
                }
            }
        }
        
        void Shuffle(){
            srand(time(NULL));
            // loop to shuffle cards
            for (int i = cards.size() - 1; i >= 0; i--){
                int j = rand() % (i + 1);
                swap(cards[i], cards[j]);
            }
            // second time for fun
            for (int i = cards.size() - 1; i >= 0; i--){
                int j = rand() % (i + 1);
                swap(cards[i], cards[j]);
            }
        }
        
        // print the letter of cards in deck
        void PrintCards(){
            if (cards.empty()) return;
            for (int i = 0; i < cards.size(); i++){
                cout << cards[i].cardLetter << endl;
            }
        }
        
        int getSizeOfDeck(){
            return cards.size();
        }
        
        Card Deal(){
            // give out cards
            Card toDeal;
            toDeal = cards.back();
            cards.pop_back();
            return toDeal;
        }
};

class Hand{
    
    // int for total value of hand
    int handSum;
    bool isHouse = false;
    bool isHouseStartingHand = false;
    // dynamic arrays for splits
    vector<Card> cardsInHand;
    vector<Card> cardsInHand2;
    vector<Card> cardsInHand3;
    
    public:
        Hand(bool isTheHouse){
            // initialize hand to 0
            handSum = 0;
            isHouse = isTheHouse;
            isHouseStartingHand = isTheHouse;
        }
        
        void DisplayHand(){
            if (cardsInHand.empty()) return;
            if (isHouseStartingHand) {
                cout << cardsInHand[0].cardLetter << " ?";
                return;
            }
            for (int i = 0; i < cardsInHand.size(); i++){
                cout << cardsInHand[i].cardLetter << " ";
            }
        }
        
        void HitMe(Deck &deck){
            // need to add later to grab from other deck
            if (deck.getSizeOfDeck() == 0) return;
            // add card to cards in hand
            cardsInHand.push_back(deck.Deal());
            if (isHouseStartingHand){
                handSum = cardsInHand[0].value;
                return;
            }
            // reset value of hand
            handSum = 0;
            // calculate value in hand
            for (int i = 0; i < cardsInHand.size(); i++){
                handSum += cardsInHand[i].value;
            }
        }
        
        void Stay(){
            // pass turn until hand finishes
        }
        
        void Split(){
            // take two of the same card and turn it into two seperate hands
        }
        
        int getHandSum(){
            return handSum;
        }
};

void ShowTable(Hand &playerHand, Hand &houseHand){
    cout << setw(17) << "House: " << houseHand.getHandSum() << endl;
    cout << setw(30) << setfill('#');
    cout << "#";
    cout << "\n";
    cout << setw(13) << setfill(' ');
    houseHand.DisplayHand();
    cout << "\n\n\n";
    cout << setw(13) << setfill(' ');
    playerHand.DisplayHand();
    cout << "\n";
    cout << setw(30) << setfill('#');
    cout << "#" << endl;
    cout << setw(18) << setfill(' ');
    cout << "Player: " << playerHand.getHandSum() << endl;
}

bool CheckForWin(Hand playerHand, Hand houseHand){
    if (playerHand.getHandSum() < 21 && playerHand.getHandSum() > houseHand.getHandSum()){
        cout << "YOU WIN!!!\n\n";
        return false;
    }
    else if (playerHand.getHandSum() > 21){
        cout << "You lose.\n\n";
        return false;
    }
    else if (houseHand.getHandSum() > 21){
        cout << "YOU WIN!!!\n\n";
        return false;
    }
    return true;
}

            switch (com){
                case 'h':
                    // need to check if deck 1 is empty
                    if (!deck1.empty()){
                        playerHand.HitMe(deck1);
                    } 
                    else {
                        playerHand.HitMe(deck2);
                    }
                    break;
                case 's':
                    playerHand.Stay();

                case 'p':

                    break;
